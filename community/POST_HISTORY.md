# 用法：bash scripts/record_community_post.sh "帖子标题" "帖子链接" "板块"

POST_TITLE="$1"
POST_URL="$2"
POST_BOARD="$3"
DATE=$(date +%Y-%m-%d)

git add community/POST_HISTORY.md
