## 📋 概述

Φ计算有三个层级，平衡速度与精度：

| 层级 | 方法 | 复杂度 | 精度 | 适用场景 |
|------|------|--------|------|----------|
| 1 | 快速估计 | O(n) | ±30% | 实时监测、大系统 |
| 2 | 标准计算 | O(n²) | ±10% | 研究应用、AI 评估 |
| 3 | 精确计算 | O(2^n) | ±1% | 小系统、理论验证 |

## 🔢 层级 3: 精确计算

### 算法

```python
import numpy as np
from itertools import partitions

def compute_phi_full(system):
    """
    精确计算Φ，复杂度 O(2^n)
    
    参数:
        system: 系统对象，包含因果图和条件概率
    
    返回:
        phi: Φ值
        mip: 最小信息分割
    """
    n = system.num_elements
    
    # 1. 计算完整因果模型信息量
    I_whole = compute_causal_info(system)
    
    # 2. 遍历所有分割，找到 MIP
    min_partition_info = float('inf')
    best_partition = None
    
    for P in all_partitions(n):  # Bell 数 B_n 个分割
        I_part = compute_partition_info(system, P)
        
        if I_part < min_partition_info:
            min_partition_info = I_part
            best_partition = P
    
    # 3. 计算Φ
    phi = I_whole - min_partition_info
    
    return phi, best_partition


def compute_causal_info(system):
    """计算因果模型信息量 I_causal"""
    # I(G): 因果图结构信息
    I_structure = np.log2(count_causal_graphs(system.num_elements))
    
    # I(CPD): 条件概率分布信息
    I_mechanism = 0
    for node in system.nodes:
        parents = system.get_parents(node)
        I_mechanism += conditional_entropy(node, parents)
    
    return I_structure + I_mechanism


def compute_partition_info(system, partition):
    """计算分割后的信息量"""
    I_total = 0
    
    # 各部分内部信息
    for part in partition:
        subsystem = system.get_subsystem(part)
        I_total += compute_causal_info(subsystem)
    
    # 部分间交互信息 (分割后剩余的)
    I_interactions = compute_inter_partition_info(system, partition)
    I_total += I_interactions
    
    return I_total
```

## 📐 层级 2: 标准计算

### 算法 (谱分割近似)

```python
import numpy as np
from scipy.linalg import eigh

def compute_phi_standard(system):
    """
    标准计算Φ，复杂度 O(n²)
    
    使用谱分析近似 MIP
    """
    n = system.num_elements
    
    # 1. 计算互信息矩阵 (O(n²))
    M = np.zeros((n, n))
    for i in range(n):
        for j in range(i, n):
            if i == j:
                M[i, j] = system.entropy(i)
            else:
                M[i, j] = mutual_info(system, i, j)
                M[j, i] = M[i, j]
    
    # 2. 归一化互信息矩阵 (O(n²))
    D = np.diag(np.sqrt(np.diag(M)))
    D_inv = np.linalg.inv(D)
    NMI = D_inv @ M @ D_inv  # NMI_ij ∈ [0, 1]
    
    # 3. 计算拉普拉斯矩阵 (O(n²))
    D_degree = np.diag(np.sum(NMI, axis=1))
    L = D_degree - NMI  # 组合拉普拉斯
    
    # 4. 谱分解 (O(n²))
    eigenvalues, eigenvectors = eigh(L)
    
    # 5. Fiedler 向量分割 (O(n))
    fiedler = eigenvectors[:, 1]  # 第二小特征值对应向量
    partition = [np.where(fiedler >= 0)[0], np.where(fiedler < 0)[0]]
    
    # 6. 计算Φ (O(n²))
    I_whole = compute_causal_info(system)
    I_part = compute_partition_info(system, partition)
    phi = I_whole - I_part
    
    return phi, partition
```

### 复杂度分析

```
主要步骤:
1. 互信息矩阵：O(n²) × T (T = 时间序列长度)
2. 归一化：O(n²)
3. 拉普拉斯：O(n²)
4. 谱分解：O(n²) (使用迭代法可 O(n·k)，k = 特征值数)
5. 分割：O(n)
6. Φ计算：O(n²)

总复杂度：O(n² × T)

实际限制：
- n < 100: 秒级
- n < 1000: 分钟级
- n < 10000: 小时级
- n ≥ 10000: 需要分布式计算
```

```python
def estimate_phi_error(system):
    """
    估计标准计算的误差
    
    使用 Bootstrap 方法
    """
    phi_estimates = []
    
    for _ in range(100):
        # 重采样时间序列
        system_boot = system.resample()
        phi, _ = compute_phi_standard(system_boot)
        phi_estimates.append(phi)
    
    phi_mean = np.mean(phi_estimates)
    phi_std = np.std(phi_estimates)
    
    return phi_mean, phi_std  # φ ± σ
```

## ⚡ 层级 1: 快速估计

### 算法

```python
import numpy as np

def compute_phi_fast(system):
    """
    快速估计Φ，复杂度 O(n)
    
    基于宏观统计量
    """
    n = system.num_elements
    
    # 1. 信息输入速率 (O(n))
    I_input = estimate_info_input_rate(system)
    
    # 2. 交互密度 (O(n))
    num_edges = system.count_edges()
    D_interaction = num_edges / n
    
    # 3. 反馈复杂度 (O(n))
    num_loops = count_feedback_loops(system)
    C_feedback = np.log2(1 + num_loops)
    
    # 4. 非线性因子 (O(n))
    N_nonlinear = estimate_nonlinearity(system)
    
    # 5. 快速估计
    phi_fast = I_input * D_interaction * C_feedback * N_nonlinear
    
    # 6. 校准 (基于回归模型)
    phi_calibrated = calibrate_phi(phi_fast, system.type)
    
    return phi_calibrated


def estimate_info_input_rate(system):
    """估计信息输入速率 (bits/s)"""
    total_entropy_rate = 0
    for node in system.nodes:
        total_entropy_rate += node.entropy_rate()
    return total_entropy_rate


def count_feedback_loops(system):
    """
    计算反馈回路数
    
    使用 DFS 检测环
    """
    # 简化：只计长度 ≤ 3 的环
    count = 0
    for node in system.nodes:
        count += count_short_loops(system, node, max_length=3)
    return count // 2  # 每个环被计数两次


def estimate_nonlinearity(system):
    """
    估计交互非线性程度
    
    基于预测误差：线性模型 vs 实际
    """
    linear_error = 0
    actual_error = 0
    
    for node in system.nodes:
        parents = system.get_parents(node)
        
        # 线性预测
        linear_pred = linear_predict(node, parents)
        linear_error += mse(node.actual, linear_pred)
        
        # 实际 (使用完整模型)
        actual_error += system.get_prediction_error(node)
    
    # 非线性程度 = 线性误差 / 实际误差
    if actual_error == 0:
        return 1.0
    
    nonlinearity = linear_error / actual_error
    return np.clip(nonlinearity, 1.0, 10.0)  # 限制范围
```

### 校准模型

```python
from sklearn.linear_model import Ridge

class PhiCalibrator:
    """Φ校准器"""
    
    def __init__(self):
        self.models = {}  # 按系统类型
    
    def train(self, systems_true, systems_fast):
        """
        使用精确Φ训练校准模型
        
        参数:
            systems_true: (系统，精确Φ) 列表
            systems_fast: (系统，快速Φ) 列表
        """
        for system_type in ['neural', 'ai', 'social', 'other']:
            # 收集该类型的系统
            X = [phi_fast for sys, phi_fast in systems_fast 
                 if sys.type == system_type]
            y = [phi_true for sys, phi_true in systems_true 
                 if sys.type == system_type]
            
            # 训练回归模型
            model = Ridge(alpha=1.0)
            model.fit(np.array(X).reshape(-1, 1), y)
            self.models[system_type] = model
    
    def calibrate(self, phi_fast, system_type):
        """校准快速Φ"""
        if system_type not in self.models:
            system_type = 'other'
        
        model = self.models[system_type]
        phi_calibrated = model.predict([[phi_fast]])[0]
        
        return max(0, phi_calibrated)  # Φ ≥ 0
```

### 复杂度分析

```
主要步骤:
1. 信息输入速率：O(n) × T
2. 交互密度：O(n) (如果边数已缓存)
3. 反馈复杂度：O(n) (近似，只计短环)
4. 非线性：O(n) × T
5. 校准：O(1)

总复杂度：O(n × T)

实际速度:
- n < 1000: 毫秒级
- n < 10000: 秒级
- n < 100000: 分钟级
- n ≥ 100000: 可行
```

```python
def mutual_info(system, i, j):
    """
    估计元素 i 和 j 的互信息

    对于连续变量：离散化或核密度估计
    """
    x_i = system.time_series[i]
    x_j = system.time_series[j]
```

### 条件熵估计

```python
def conditional_entropy(X, Y):
    """
    H(X | Y) = H(X, Y) - H(Y)
    """
    h_xy = joint_entropy(X, Y)
    h_y = entropy(Y)
    return h_xy - h_y


def entropy(x, bins=10):
    """Shannon 熵"""
    hist, _ = np.histogram(x, bins=bins, density=True)
    hist = hist[hist > 0]  # 移除零
    return -np.sum(hist * np.log2(hist)) / bins


def joint_entropy(x, y, bins=10):
    """联合熵"""
    hist, _, _ = np.histogram2d(x, y, bins=bins, density=True)
    hist = hist[hist > 0]
    return -np.sum(hist * np.log2(hist)) / (bins * bins)
```
