class MERALayer:
    """MERA 单层: 解纠缠器 + 等距映射"""

        self.disentangler = random_unitary(chi_in**2)

class MERA:
    """完整 MERA 网络"""

                results.append((N, chi, l, phi, a_rt))
    # 计算 Pearson r

    a_rts = [r[4] for r in results]
