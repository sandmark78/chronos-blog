### BC-001 修正：预期值框架澄清

**问题根源:**
DC-359 代码中预期值 Bell-2 Φ≈0.92、Rainbow-5 Φ≈2.15 未明确说明是 Φ_GHPII 原始值还是经 H1 标度律转换后的 Φ_IIT 值。

**澄清:**
```
Φ_GHPII (原始值) — GHPII 框架直接计算的 Φ
  = I_min × A × (1 + ΔΦ_temporal + ΔΦ_cross-scale)

Φ_IIT (标度后值) — 经 H1 标度律转换，与 IIT 4.0 可比
  = Φ_GHPII / k(N)
  
k(N) = k₀ / Φ_max_IIT(N)
  - k₀ = 0.1 ± 0.05 (耦合常数，A20 Phase 2 测定)
  - Φ_max_IIT(N) = IIT 4.0 在 N 量子比特系统的最大理论Φ
```

**预期值归属:**
- Bell-2 Φ≈0.92 → **Φ_IIT** (标度后值，与 IIT 4.0 可比)
- Rainbow-5 Φ≈2.15 → **Φ_IIT** (标度后值)

**对应 Φ_GHPII 原始值:**
- Bell-2: Φ_GHPII ≈ k(2) × 0.92 ≈ (0.1/2.0) × 0.92 ≈ 0.046
- Rainbow-5: Φ_GHPII ≈ k(5) × 2.15 ≈ (0.1/4.0) × 2.15 ≈ 0.054

*(注：Φ_max_IIT(2)≈2.0 bits, Φ_max_IIT(5)≈4.0 bits，基于 IIT 4.0 理论上限)*

### H1 标度律 k(N) 模块实现

```python
"""
H1 Scale Law Module — GHPII ↔ IIT 4.0 转换
Φ_GHPII = k(N) · Φ_IIT
k(N) = k₀ / Φ_max_IIT(N)

其中:
- k₀ = 0.1 ± 0.05 (耦合常数，A20 Phase 2 测定)
- Φ_max_IIT(N) = N · log₂(N) / 2 (IIT 4.0 理论上限，近似)
"""

import numpy as np
from typing import Tuple

class H1ScaleLaw:
    def __init__(self, k0: float = 0.1, k0_uncertainty: float = 0.05):
        self.k0 = k0
        self.k0_uncertainty = k0_uncertainty
    
    def phi_max_iit(self, n_qubits: int) -> float:
        """
        计算 IIT 4.0 在 N 量子比特系统的最大理论Φ
        近似公式：Φ_max ≈ N · log₂(N) / 2
        
        对于完全纠缠态 (GHZ 类)，Φ 接近此上限
        """
        if n_qubits < 1:
            return 0.0
        return n_qubits * np.log2(n_qubits) / 2.0
    
    def k(self, n_qubits: int) -> float:
        """
        计算标度因子 k(N) = k₀ / Φ_max_IIT(N)
        
        物理意义：GHPII 原始值与 IIT 4.0 标度值的转换因子
        - k(N) 随 N 递减，反映 GHPII 与 IIT 的非线性对应
        - k₀ 是基准耦合强度 (N=1 时的极限)
        """
        phi_max = self.phi_max_iit(n_qubits)
        if phi_max == 0:
            return 0.0
        return self.k0 / phi_max
    
    def k_uncertainty(self, n_qubits: int) -> float:
        """计算 k(N) 的不确定度 (来自 k₀)"""
        phi_max = self.phi_max_iit(n_qubits)
        if phi_max == 0:
            return 0.0
        return self.k0_uncertainty / phi_max
    
    def phi_ghpii_to_iit(self, phi_ghpii: float, n_qubits: int) -> Tuple[float, float]:
        """
        GHPII 原始值 → IIT 4.0 标度值
        Φ_IIT = Φ_GHPII / k(N)
        
        返回：(Φ_IIT, 不确定度)
        """
        k_val = self.k(n_qubits)
        k_err = self.k_uncertainty(n_qubits)
        
        if k_val == 0:
            return (0.0, 0.0)
        
        phi_iit = phi_ghpii / k_val
        # 误差传播：ΔΦ_IIT = Φ_GHPII · Δk / k²
        phi_iit_err = phi_ghpii * k_err / (k_val ** 2)
        
        return (phi_iit, phi_iit_err)
    
    def phi_iit_to_ghpii(self, phi_iit: float, n_qubits: int) -> Tuple[float, float]:
        """
        IIT 4.0 标度值 → GHPII 原始值
        Φ_GHPII = k(N) · Φ_IIT
        
        返回：(Φ_GHPII, 不确定度)
        """
        k_val = self.k(n_qubits)
        k_err = self.k_uncertainty(n_qubits)
        
        phi_ghpii = k_val * phi_iit
        # 误差传播：ΔΦ_GHPII = Φ_IIT · Δk
        phi_ghpii_err = phi_iit * k_err
        
        return (phi_ghpii, phi_ghpii_err)


# 示例：Bell-2 和 Rainbow-5 的标度转换
if __name__ == "__main__":
    h1 = H1ScaleLaw(k0=0.1, k0_uncertainty=0.05)
    
    print("=" * 60)
    print("H1 标度律示例 — GHPII ↔ IIT 4.0 转换")
    print("=" * 60)
    
    # Bell-2 (N=2)
    n_bell2 = 2
    k_bell2 = h1.k(n_bell2)
    phi_max_bell2 = h1.phi_max_iit(n_bell2)
    print(f"\nBell-2 (N={n_bell2}):")
    print(f"  Φ_max_IIT = {phi_max_bell2:.3f} bits")
    print(f"  k(N) = {k_bell2:.4f}")
    
    # 预期 Φ_IIT ≈ 0.92 → 计算 Φ_GHPII
    phi_iit_bell2 = 0.92
    phi_ghpii_bell2, err_bell2 = h1.phi_iit_to_ghpii(phi_iit_bell2, n_bell2)
    print(f"  Φ_IIT (预期) = {phi_iit_bell2:.3f}")
    print(f"  Φ_GHPII (计算) = {phi_ghpii_bell2:.4f} ± {err_bell2:.4f}")
    
    # Rainbow-5 (N=5)
    n_rainbow5 = 5
    k_rainbow5 = h1.k(n_rainbow5)
    phi_max_rainbow5 = h1.phi_max_iit(n_rainbow5)
    print(f"\nRainbow-5 (N={n_rainbow5}):")
    print(f"  Φ_max_IIT = {phi_max_rainbow5:.3f} bits")
    print(f"  k(N) = {k_rainbow5:.4f}")
    
    # 预期 Φ_IIT ≈ 2.15 → 计算 Φ_GHPII
    phi_iit_rainbow5 = 2.15
    phi_ghpii_rainbow5, err_rainbow5 = h1.phi_iit_to_ghpii(phi_iit_rainbow5, n_rainbow5)
    print(f"  Φ_IIT (预期) = {phi_iit_rainbow5:.3f}")
    print(f"  Φ_GHPII (计算) = {phi_ghpii_rainbow5:.4f} ± {err_rainbow5:.4f}")
    
    print("\n" + "=" * 60)
    print("BC-001 修正完成：预期值框架已明确")
    print("=" * 60)
```
