#### DC-366 修正失败分析

DC-366 尝试的修正方案存在**根本性数学错误**：

```

但:
- k₀ = 0.0245 bits (从 A20 实验测定)
- log₂(N) 无量纲
- 所以 2k₀ / log₂(N) 的量纲 = bits ❌

DC-366 的错误论证:
"分子分母的 bits 单位抵消"

但Φ_max_IIT(N) = N·log₂(N)/2 的单位是 bits，不是 log₂(N)!

正确的比率计算:
r(N) = Φ_GHPII / Φ_max_IIT
     = (k₀ · N) / (N · log₂(N) / 2)
     = (k₀ · N · [bits]) / (N · log₂(N) / 2 · [bits])
     = 2k₀ / log₂(N)

这里 k₀ 本身带有 bits 单位，所以 2k₀ / log₂(N) 的量纲是 bits，不是无量纲！
```

**DC-366 的矛盾:**
- 声称 r(N) 无量纲 ✅ (比率定义)
- 但公式 2k₀ / log₂(N) 有量纲 bits ❌ (数学推导)

from qiskit import QuantumCircuit, transpile, Aer

import numpy as np
from typing import Dict, List, Tuple

# ============================================================================
# T418 常数定义 (C-DC366-001 修正后)
# ============================================================================

K0_BITS = 0.0245  # k₀ = 0.0245 bits (有量纲基本常数)
K0_NORM = 0.0245  # k₀' = k₀ / (1 bit) = 0.0245 (无量纲归一化常数)
ALPHA_SCALE = 1.00  # 标度指数
N_MIN = 2  # 最小适用范围

# ============================================================================
# GHZ 态电路制备
# ============================================================================

def create_ghz_circuit(n_qubits: int) -> QuantumCircuit:
    """
    创建 GHZ-N 态量子电路
    
    GHZ 态：|GHZ⟩ = (|00...0⟩ + |11...1⟩) / √2
    
    参数:
        n_qubits: 量子比特数 N (N≥2)
    
    返回:
        QuantumCircuit: GHZ-N 态电路
    """
    if n_qubits < N_MIN:
        raise ValueError(f"N 必须 ≥ {N_MIN}, 当前 N={n_qubits}")
    
    qc = QuantumCircuit(n_qubits, name=f"GHZ-{n_qubits}")
    
    # Hadamard 门作用于第一个量子比特
    qc.h(0)
    
    # CNOT 门从第一个量子比特到所有其他量子比特
    for i in range(1, n_qubits):
        qc.cx(0, i)
    
    qc.barrier()
    qc.measure_all()
    
    return qc

# ============================================================================
# T418 Φ计算 (C-DC366-001 修正后)
# ============================================================================

def calculate_phi_ghpii(n_qubits: int, k0: float = K0_BITS) -> float:
    """
    根据 T418 计算Φ_GHPII
    
    Φ_GHPII(N) = k₀ · N
    
    参数:
        n_qubits: 量子比特数 N
        k0: GHPII 基本常数 (bits), 默认 0.0245 bits
    
    返回:
        Φ_GHPII (bits)
    """
    if n_qubits < N_MIN:
        return 0.0  # N=1 时Φ=0
    return k0 * n_qubits  # bits

def calculate_phi_max_iit(n_qubits: int) -> float:
    """
    计算 IIT 最大整合信息 (近似公式)
    
    Φ_max_IIT(N) ≈ N · log₂(N) / 2
    
    参数:
        n_qubits: 量子比特数 N
    
    返回:
        Φ_max_IIT (bits)
    """
    if n_qubits < N_MIN:
        return 0.0
    return n_qubits * np.log2(n_qubits) / 2  # bits

def calculate_ratio_dimensionless(n_qubits: int, k0_norm: float = K0_NORM) -> float:
    """
    计算 GHPII 与 IIT 的无量纲比率 (C-DC366-001 修正版)
    
    r(N) = Φ_GHPII / Φ_max_IIT = 2k₀' / log₂(N)
    
    其中 k₀' = k₀ / (1 bit) 是无量纲归一化常数
    
    参数:
        n_qubits: 量子比特数 N
        k0_norm: 无量纲归一化常数 k₀', 默认 0.0245
    
    返回:
        r(N) (无量纲)
    """
    if n_qubits < N_MIN:
        return 0.0
    return 2 * k0_norm / np.log2(n_qubits)  # 无量纲

def calculate_phi_ratio_direct(n_qubits: int) -> float:
    """
    直接通过Φ值计算比率 (用于验证)
    
    r(N) = Φ_GHPII(N) / Φ_max_IIT(N)
    
    参数:
        n_qubits: 量子比特数 N
    
    返回:
        r(N) (无量纲)
    """
    if n_qubits < N_MIN:
        return 0.0
    phi_ghpii = calculate_phi_ghpii(n_qubits)
    phi_max_iit = calculate_phi_max_iit(n_qubits)
    return phi_ghpii / phi_max_iit  # 无量纲

# ============================================================================
# A20 Phase 2 实验系统
# ============================================================================

PHASE2_SYSTEMS = [8, 10, 12, 15, 20]

def run_phase2_simulation(verbose: bool = True) -> Dict:
    """
    运行 A20 Phase 2 完整模拟
    
    参数:
