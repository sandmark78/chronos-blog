#### 2.1.1 Qiskit 实现

```python
import numpy as np
from qiskit import QuantumCircuit, Aer, execute
from qiskit.providers.aer.noise import NoiseModel
from qiskit.providers.aer.noise.errors import depolarizing_error, thermal_relaxation_error

# 参数设置
k0 = 0.0300  # bits
N_c_initial = 68  # 初始估计

# IBM Brisbane 噪声参数
t1 = 100e-6  # 100 μs
t2 = 80e-6   # 80 μs
gate_time_1q = 50e-9   # 50 ns
gate_time_2q = 200e-9  # 200 ns
error_1q = 0.0005  # 0.05%
error_2q = 0.005   # 0.5%
readout_error = 0.01  # 1%

def create_noise_model():
    """创建 IBM Brisbane 风格噪声模型"""
    noise_model = NoiseModel()
    
    # 单比特门错误
    error_1q_dep = depolarizing_error(error_1q, 1)
    noise_model.add_all_qubit_quantum_error(error_1q_dep, ['u1', 'u2', 'u3'])
    
    # 双比特门错误
    error_2q_dep = depolarizing_error(error_2q, 2)
    noise_model.add_all_qubit_quantum_error(error_2q_dep, ['cx'])
    
    # 弛豫错误
    error_t1_1q = thermal_relaxation_error(t1, t2, gate_time_1q)
    error_t1_2q = thermal_relaxation_error(t1, t2, gate_time_2q)
    noise_model.add_all_qubit_quantum_error(error_t1_1q, ['u1', 'u2', 'u3'])
    noise_model.add_all_qubit_quantum_error(error_t1_2q, ['cx'])
    
    return noise_model

def ghz_circuit(n_qubits):
    """制备 GHZ 态"""
    qc = QuantumCircuit(n_qubits, n_qubits)
    qc.h(0)
    for i in range(1, n_qubits):
        qc.cx(0, i)
    qc.measure_all()
    return qc

def calculate_phi_from_counts(counts, n_qubits):
    """从测量计数计算Φ_GHPII"""
    total = sum(counts.values())
    
    # 提取 |00...0⟩ 和 |11...1⟩ 计数
    key_0 = '0' * n_qubits
    key_1 = '1' * n_qubits
    
    count_0 = counts.get(key_0, 0)
    count_1 = counts.get(key_1, 0)
    
    # 计算 GHZ 子空间占比
    p_ghz = (count_0 + count_1) / total
    
    # 计算相干性 (简化模型)
    # 理想 GHZ: p_ghz = 1, coherence = 1
    # 退相干后：p_ghz < 1, coherence ≈ p_ghz
    coherence = p_ghz
    
    # Φ_GHPII = k₀ × (N-1) × coherence
    phi = k0 * (n_qubits - 1) * coherence
    
    return phi

def run_simulation(n_qubits, shots=1000):
    """运行单次模拟"""
    qc = ghz_circuit(n_qubits)
    noise_model = create_noise_model()
    backend = Aer.get_backend('qasm_simulator')
    job = execute(qc, backend, shots=shots, noise_model=noise_model)
    result = job.result()
    counts = result.get_counts()
    phi = calculate_phi_from_counts(counts, n_qubits)
    return phi

# 执行测量
measurement_points = [50, 60, 70, 80]
results = {}

for n in measurement_points:
    phi_values = []
    for _ in range(1000):  # 1000 次测量
        phi = run_simulation(n, shots=1024)
        phi_values.append(phi)
    
    mean_phi = np.mean(phi_values)
    std_phi = np.std(phi_values) / np.sqrt(len(phi_values))  # 标准误差
    results[n] = {'mean': mean_phi, 'std': std_phi, 'n_measurements': 1000}
    print(f"N={n}: Φ = {mean_phi:.4f} ± {std_phi:.4f} bits")
```

### 2.2 T420 更新 (统一 N_c 定义)

**原 T420 (DC-387):**
```
N_c = 1/(γ_single × τ_0)
```

**更新后 T420 v1.1:**
```
N_c = ln(2)/(γ × τ_gate)

其中：
- γ = 有效退相干率 (包含门错误贡献)
- τ_gate = 量子门时间 (CNOT ~200ns)
- ln(2) ≈ 0.693 来自保真度阈值 F=1/2

物理推导：
当 N > N_c 时，退相干累积使 GHZ 态保真度降至 F < 1/2
F(N) ≈ exp(-N × γ × τ_gate)
令 F(N_c) = 1/2:
  exp(-N_c × γ × τ_gate) = 1/2
  -N_c × γ × τ_gate = ln(1/2) = -ln(2)
  N_c = ln(2)/(γ × τ_gate)
```

**与 T421 的关系:**
- T420 v1.1 与 T421 v1.0 **完全一致**
- C-391-A01 🟡 解决 ✅
