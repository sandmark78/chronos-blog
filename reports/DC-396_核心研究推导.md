### 2.1 实验设计

**目标:** 高精度测量Φ_max，验证 T420 饱和行为预测

**系统:** GHZ-100/150/200/300/500 (5 点，延伸至 N=500)

**测量方法:**
```
Φ_GHPII(N) = k₀ × S_GHPII(N)
S_GHPII(N) = -Tr(ρ_log₂ ρ) + Tr(ρ_diag log₂ ρ_diag)
```

**噪声模型:** IBM Brisbane 典型参数
- T₁ = 100 μs
- T₂ = 80 μs
- 单比特门错误率: 0.05%
- 双比特门错误率: 0.5%
- 读取错误率: 2%

**采样:** 每点 10000 shots (统计误差<1%)

### 2.2 Qiskit 实现

```python
# A20_Phase6C_PhiMax_Measurement.py
# DC-396 核心执行代码

from qiskit import QuantumCircuit, Aer, execute
from qiskit.providers.aer.noise import NoiseModel
import numpy as np

def create_ghz_state(n_qubits):
    """创建 N 量子比特 GHZ 态"""
    qc = QuantumCircuit(n_qubits)
    qc.h(0)
    for i in range(1, n_qubits):
        qc.cx(0, i)
    return qc

def compute_phi_ghpii(circuit, noise_model, shots=10000):
    """计算Φ_GHPII = k₀ × S_GHPII"""
    backend = Aer.get_backend('qasm_simulator')
    backend.set_options(noise_model=noise_model)
    
    # 添加测量
    qc_with_meas = circuit.copy()
    qc_with_meas.measure_all()
    
    # 执行
    result = execute(qc_with_meas, backend, shots=shots).result()
    counts = result.get_counts()
    
    # 计算香农熵 (对角部分)
    p_diag = np.array(list(counts.values())) / shots
    h_diag = -np.sum(p_diag * np.log2(p_diag + 1e-10))
    
    # 计算冯诺依曼熵 (需要态层析，简化为纯态近似)
    # 对于理想 GHZ: S_vN = 0
    # 对于噪声 GHZ: S_vN ≈ f(noise_params)
    s_vn = estimate_von_neumann_entropy(circuit, noise_model)
    
    # S_GHPII = H_diag - S_vN
    s_ghpii = h_diag - s_vn
    
    # Φ_GHPII = k₀ × S_GHPII (k₀ = 0.0300 bits)
    k0 = 0.0300
    phi_ghpii = k0 * s_ghpii
    
    return phi_ghpii, s_ghpii, h_diag, s_vn

def estimate_von_neumann_entropy(circuit, noise_model):
    """估计冯诺依曼熵 (简化模型)"""
    # 基于噪声参数的近似
    # 详细实现需要态层析，此处用经验公式
    n = circuit.num_qubits
    error_rate = 0.005  # 平均门错误率
    
    # 经验公式: S_vN ≈ n × error_rate × log₂(n)
    s_vn = n * error_rate * np.log2(n + 1)
    return s_vn

# 执行 Phase 6C 测量
n_values = [100, 150, 200, 300, 500]
results = []

for n in n_values:
    print(f"测量 N={n}...")
    ghz_circuit = create_ghz_state(n)
    phi, s_ghpii, h_diag, s_vn = compute_phi_ghpii(ghz_circuit, noise_model)
    results.append({
        'N': n,
        'Φ': phi,
        'S_GHPII': s_ghpii,
        'H_diag': h_diag,
        'S_vN': s_vn
    })
    print(f"  Φ({n}) = {phi:.4f} bits")
```
