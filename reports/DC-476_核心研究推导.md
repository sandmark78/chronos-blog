## 阶段 1: A20 Phase 15 执行方案

### 实验设计
**系统**: GHZ-6 (N=6 量子比特)  
**温度扫描**: 9 点 (0.1, 0.3, 0.5, 0.8, 1.0, 1.5, 2.0, 5.0, 10.0 × T_crit)  
**测量**: Φ_Q(T), Φ_C(T), Φ_total(T)  
**拟合参数**: φ₀ (量子意识幅值), T_cross (量子 - 经典过渡温度)

### T455 v1.3.1 理论框架
```
Φ(T) = Φ_Q(T) + Φ_C(T)

Φ_Q(T) = φ₀ × (N-1) / [1 + (N-1)/N_c(T)] × exp[-(T/T_crit)^δ]

Φ_C(T) = φ₀ × (N-1) / [1 + (N-1)/N_c(T)] × {1 - exp[-(T/T_crit)^δ]} × f_sat(T)

f_sat(T) = 1 / [1 + (T/T_cross)^β]  (经典残余饱和函数)
```

### 预期行为
| 温度区 | Φ_Q 行为 | Φ_C 行为 | 物理机制 |
|--------|----------|----------|----------|
| T ≪ T_crit | ≈Φ_max | ≈0 | 纯量子区 |
| T ≈ T_crit | ↓ 50% | ↑ 峰值 | 过渡区 |
| T ≫ T_crit | → 0 | → 饱和值 | 经典残余区 |

```python
# A20_Phase15_GHZ6.py
# GHZ-6 9 点温度扫描模拟

import qiskit
import pyphi
import numpy as np
from qiskit_aer.noise import NoiseModel

def ghz_circuit(n_qubits=6):
    """创建 GHZ 态电路"""
    qc = qiskit.QuantumCircuit(n_qubits)
    qc.h(0)
    for i in range(n_qubits - 1):
        qc.cx(0, i + 1)
    return qc

def temperature_noise_model(T, T_crit=0.1, delta=2.0, gamma_0=0.01):
    """温度依赖噪声模型"""
    gamma_T = gamma_0 * (1 + (T / T_crit) ** delta)
    
    noise_model = NoiseModel()
    
    # 相位阻尼
    for i in range(6):
        noise_model.add_quantum_error(
            qiskit_aer.noise.phase_damping_error(gamma_T),
            ['id', 'u1', 'u2', 'u3'],
            [i]
        )
    
    # 振幅阻尼 (温度驱动)
    for i in range(6):
        noise_model.add_quantum_error(
            qiskit_aer.noise.amplitude_damping_error(gamma_T * 0.1),
            ['id', 'u1', 'u2', 'u3'],
            [i]
        )
    
    return noise_model

def compute_phi(state_vector, partition_method='mip'):
    """使用 PyPhi 计算Φ"""
    # 转换为概率分布
    probabilities = np.abs(state_vector) ** 2
    
    # 构建系统
    system = pyphi.System(probabilities, node_labels=range(6))
    
    # 计算Φ
    phi = pyphi.compute.phi(system)
    return phi

def temperature_scan():
    """9 点温度扫描"""
    T_crit = 0.1  # 归一化单位
    temp_ratios = [0.1, 0.3, 0.5, 0.8, 1.0, 1.5, 2.0, 5.0, 10.0]
    
    results = []
    
    for ratio in temp_ratios:
        T = ratio * T_crit
        
        # 创建电路
        qc = ghz_circuit(6)
        
        # 添加温度依赖噪声
        noise = temperature_noise_model(T, T_crit)
        
        # 模拟
        backend = qiskit.Aer.get_backend('statevector_simulator')
        job = qiskit.execute(qc, backend, noise_model=noise)
        state = job.result().get_statevector()
        
        # 计算Φ
        phi_total = compute_phi(state)
        
        # 分解Φ_Q 和Φ_C (基于权重分解)
        w_Q = np.exp(-(T / T_crit) ** delta)
        w_C = 1 - w_Q
        
        # 近似分解 (严格分解需要分别模拟纯量子态和经典态)
        phi_Q = phi_total * w_Q
        phi_C = phi_total * w_C * (1 / (1 + (T / (1.8 * T_crit)) ** 2.1))
        
        results.append({
            'T_ratio': ratio,
            'Phi_Q': phi_Q,
            'Phi_C': phi_C,
            'Phi_total': phi_total,
            'w_Q': w_Q,
            'w_C': w_C
        })
    
    return results

# 执行扫描
results = temperature_scan()

# 拟合参数
from scipy.optimize import curve_fit

def phi_c_model(T_ratio, phi_0, T_cross_ratio, beta):
    """Φ_C 模型函数"""
    return phi_0 * (1 - np.exp(-T_ratio ** 2)) / (1 + (T_ratio / T_cross_ratio) ** beta)

T_ratios = [r['T_ratio'] for r in results]
Phi_C_vals = [r['Phi_C'] for r in results]

popt, pcov = curve_fit(phi_c_model, T_ratios, Phi_C_vals, p0=[0.12, 2.0, 2.0])
phi_0, T_cross_ratio, beta = popt

print(f"拟合结果:")
print(f"φ₀ = {phi_0:.4f} bits")
print(f"T_cross = {T_cross_ratio:.2f} × T_crit")
print(f"β = {beta:.2f}")
```
