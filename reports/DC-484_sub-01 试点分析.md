```python
#!/usr/bin/env python3
"""
Step 1: 数据加载

import nibabel as nib
import pandas as pd
import numpy as np
from pathlib import Path

def load_fmri_data(sub_id, run_ids, data_root):
    """加载 fMRI 数据和事件文件"""
    
    fmri_data = []
    events_data = []
    
    for run in run_ids:
        # NIfTI 文件路径
        nii_path = Path(data_root) / f"sub-{sub_id}" / "func" / \
                   f"sub-{sub_id}_task-balloonanalogrisktask_run-{run:02d}_bold.nii.gz"
        
        # 事件文件路径
        events_path = Path(data_root) / f"sub-{sub_id}" / "func" / \
                      f"sub-{sub_id}_task-balloonanalogrisktask_run-{run:02d}_events.tsv"
        
        # 加载 NIfTI
        img = nib.load(str(nii_path))
        fmri = img.get_fdata()  # 4D array: (x, y, z, time)
        fmri_data.append(fmri)
        
        # 加载事件
        events = pd.read_csv(events_path, sep='\t')
        events_data.append(events)
        
        print(f"Loaded run-{run}: {fmri.shape}, {len(events)} events")
    
    return fmri_data, events_data

# 使用示例
```

#### 2.2.3 互信息矩阵计算

```python
#!/usr/bin/env python3
"""
Step 3: 互信息矩阵计算
"""
from sklearn.metrics import mutual_info_score

def compute_mi_matrix(time_series, n_bins=10):
    """计算 ROI 间互信息矩阵"""
    
    n_timepoints, n_rois = time_series.shape
    mi_matrix = np.zeros((n_rois, n_rois))
    
    # 离散化时间序列
    discretized = np.zeros_like(time_series, dtype=int)
    for roi in range(n_rois):
        ts = time_series[:, roi]
        bins = np.linspace(ts.min(), ts.max(), n_bins + 1)
        discretized[:, roi] = np.digitize(ts, bins) - 1
    
    # 计算互信息
    for i in range(n_rois):
        for j in range(i + 1, n_rois):
            mi = mutual_info_score(discretized[:, i], discretized[:, j])
            mi_matrix[i, j] = mi
            mi_matrix[j, i] = mi
    
    return mi_matrix

# 对 3 个 run 分别计算，然后平均
mi_matrices = [compute_mi_matrix(ts) for ts in time_series]
mi_matrix_avg = np.mean(mi_matrices, axis=0)

print(f"MI matrix shape: {mi_matrix_avg.shape}")
print(f"MI range: [{mi_matrix_avg.min():.4f}, {mi_matrix_avg.max():.4f}]")
```

#### 2.2.4 Φ计算与子系统采样

```python
#!/usr/bin/env python3
"""
Step 4: 子系统采样与Φ计算
"""

def compute_phi_simplified(mi_matrix):
    """
    计算简化版Φ (基于特征值分解)
    
    注意：这是近似方法，完整 MIP 计算需要 IIT 4.0 算法
    """
    # 特征值分解
    eigenvalues = np.linalg.eigvalsh(mi_matrix)
    
    # 正特征值之和作为Φ的代理
    phi = np.sum(eigenvalues[eigenvalues > 1e-10])
    
    return phi

def sample_subsystems(mi_matrix, n_min=2, n_max=15, n_samples=100, random_seed=42):
    """随机采样不同规模的子系统"""
    
    np.random.seed(random_seed)
    n_total = mi_matrix.shape[0]
    
    results = []
    
    for n in range(n_min, n_max + 1):
        phi_values = []
        
        for _ in range(n_samples):
            # 随机选择 n 个 ROI
            indices = np.random.choice(n_total, n, replace=False)
            sub_mi = mi_matrix[np.ix_(indices, indices)]
            
            # 计算Φ
            phi = compute_phi_simplified(sub_mi)
            phi_values.append(phi)
        
        phi_mean = np.mean(phi_values)
        phi_std = np.std(phi_values)
        
        results.append({
            'N': n,
            'phi_mean': phi_mean,
            'phi_std': phi_std,
            'phi_max_theoretical': n * np.log2(n) / 2
        })
        
        print(f"N={n:2d}: Φ={phi_mean:8.4f} ± {phi_std:.4f}, "
              f"Φ_max={n * np.log2(n) / 2:.4f}")
    
    return results

# 执行采样
sampling_results = sample_subsystems(mi_matrix_avg, n_min=2, n_max=15, n_samples=100)
```

#### 2.2.5 η_IIT 计算与模型拟合

```python
#!/usr/bin/env python3
"""
Step 5: η_IIT 计算与模型拟合
"""
from scipy.optimize import curve_fit
import matplotlib.pyplot as plt

def eta_iit_model(N, eta0, epsilon):
    """η_IIT 指数衰减模型"""
    return eta0 * np.exp(-epsilon * (N - 2))

def fit_eta_iit(sampling_results):
    """拟合η_IIT 模型"""
    
    # 提取数据
    N_values = np.array([r['N'] for r in sampling_results])
    phi_actual = np.array([r['phi_mean'] for r in sampling_results])
    phi_max = np.array([r['phi_max_theoretical'] for r in sampling_results])
    
    # 计算η_IIT
    eta_iit = phi_actual / phi_max
    
    # 拟合模型
    popt, pcov = curve_fit(
        eta_iit_model, 
        N_values, 
        eta_iit,
        p0=[1.0, 0.15],  # 初始猜测
        bounds=([0.5, 0.0], [1.5, 0.5])  # 参数边界
    )
    
    eta0_fit, epsilon_fit = popt
    eta0_std, epsilon_std = np.sqrt(np.diag(pcov))
    
    # 计算 R²
    eta_predicted = eta_iit_model(N_values, *popt)
    ss_res = np.sum((eta_iit - eta_predicted) ** 2)
    ss_tot = np.sum((eta_iit - np.mean(eta_iit)) ** 2)
    r_squared = 1 - (ss_res / ss_tot)
    
    return {
        'eta0': eta0_fit,
        'eta0_std': eta0_std,
        'epsilon': epsilon_fit,
        'epsilon_std': epsilon_std,
        'r_squared': r_squared,
        'N_values': N_values,
        'eta_iit': eta_iit,
        'eta_predicted': eta_predicted
    }

# 执行拟合
fit_results = fit_eta_iit(sampling_results)

print(f"\n{'='*50}")
print(f"η_IIT 模型拟合结果")
print(f"{'='*50}")
print(f"η₀ = {fit_results['eta0']:.4f} ± {fit_results['eta0_std']:.4f}")
print(f"ε  = {fit_results['epsilon']:.4f} ± {fit_results['epsilon_std']:.4f}")
print(f"R² = {fit_results['r_squared']:.4f}")
print(f"{'='*50}")
```
