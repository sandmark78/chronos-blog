```python
def extract_roi_timeseries(fmri_img, atlas_data, method='mean'):
    """
    从 fMRI 数据提取 ROI 时间序列
    
    参数:
        fmri_img: fMRI 4D 图像 (nibabel object)
        atlas_data: 图谱 3D 数据
        method: 'mean' | 'median' | 'pca'
    
    返回:
        roi_ts: ROI 时间序列 (n_timepoints, n_regions)
    """
    fmri_data = fmri_img.get_fdata()
    n_timepoints = fmri_data.shape[3]
    n_regions = len(np.unique(atlas_data)) - 1  # 排除 0 (背景)
    
    roi_ts = np.zeros((n_timepoints, n_regions))
    
    for i in range(1, n_regions + 1):
        mask = (atlas_data == i)
        if method == 'mean':
            roi_ts[:, i-1] = np.mean(fmri_data[mask, :], axis=0)
        elif method == 'median':
            roi_ts[:, i-1] = np.median(fmri_data[mask, :], axis=0)
        elif method == 'pca':
            # PCA 第一主成分
            from sklearn.decomposition import PCA
            pca = PCA(n_components=1)
            roi_ts[:, i-1] = pca.fit_transform(fmri_data[mask, :]).flatten()
    
    return roi_ts
```

### 修正 2: KSG 非参数 MI 估计 (实现)

**完整代码:**
```python
#!/usr/bin/env python3
"""
fMRI η_IIT 分析流程 v2.0 - KSG 非参数互信息估计
"""

import numpy as np
from sklearn.feature_selection import mutual_info_regression
from scipy.spatial import cKDTree

def compute_mi_ksg(X, Y, k=5, add_noise=1e-10):
    """
    KSG (Kraskov-Stögbauer-Grassberger) 非参数 MI 估计
    
    参数:
        X: 时间序列 (n_samples, n_features_X)
        Y: 时间序列 (n_samples, n_features_Y)
        k: 近邻数 (默认 5，建议 3-7)
        add_noise: 数值稳定性的小噪声
    
    返回:
        MI: 互信息估计 (nats)
    """
    # 添加小噪声避免数值问题
    X = X + np.random.normal(0, add_noise, X.shape)
    Y = Y + np.random.normal(0, add_noise, Y.shape)
    
    # 使用 sklearn 的 KSG 估计器
    mi = mutual_info_regression(X, Y, n_neighbors=k, random_state=42)
    
    # 从 nats 转换为 bits
    mi_bits = mi / np.log(2)
    
    return mi_bits

def compute_mi_matrix_ksg(roi_ts, k=5):
    """
    计算 ROI 间的互信息矩阵
    
    参数:
        roi_ts: ROI 时间序列 (n_timepoints, n_regions)
        k: KSG 近邻数
    
    返回:
        mi_matrix: 互信息矩阵 (n_regions, n_regions)
    """
    n_regions = roi_ts.shape[1]
    mi_matrix = np.zeros((n_regions, n_regions))
    
    for i in range(n_regions):
        for j in range(i+1, n_regions):
            mi = compute_mi_ksg(
                roi_ts[:, i:i+1],
                roi_ts[:, j:j+1],
                k=k
            )
            mi_matrix[i, j] = mi
            mi_matrix[j, i] = mi  # 对称性
    
    return mi_matrix

def compare_mi_methods(roi_ts):
    """
    对比参数化 (Gaussian) vs 非参数 (KSG) MI 估计
    
    返回:
        comparison: dict with both estimates and difference metrics
    """
    # 参数化方法 (Gaussian)
    from scipy.stats import entropy
    
    def mi_gaussian(X, Y):
        """Gaussian MI: -0.5 * log(1 - corr²)"""
        corr = np.corrcoef(X.flatten(), Y.flatten())[0, 1]
        if abs(corr) >= 1:
            return 0
        mi = -0.5 * np.log(1 - corr**2)
        return mi / np.log(2)  # bits
    
    n_regions = roi_ts.shape[1]
    mi_gaussian_matrix = np.zeros((n_regions, n_regions))
    mi_ksg_matrix = np.zeros((n_regions, n_regions))
    
    for i in range(n_regions):
        for j in range(i+1, n_regions):
            mi_g = mi_gaussian(roi_ts[:, i:i+1], roi_ts[:, j:j+1])
            mi_k = compute_mi_ksg(roi_ts[:, i:i+1], roi_ts[:, j:j+1])
            
            mi_gaussian_matrix[i, j] = mi_g
            mi_gaussian_matrix[j, i] = mi_g
            mi_ksg_matrix[i, j] = mi_k
            mi_ksg_matrix[j, i] = mi_k
    
    # 比较统计
    upper_tri_g = mi_gaussian_matrix[np.triu_indices(n_regions, k=1)]
    upper_tri_k = mi_ksg_matrix[np.triu_indices(n_regions, k=1)]
    
    comparison = {
        'mi_gaussian_mean': np.mean(upper_tri_g),
        'mi_ksg_mean': np.mean(upper_tri_k),
        'relative_difference': (upper_tri_k - upper_tri_g) / (upper_tri_g + 1e-10),
        'ksg_higher_ratio': np.mean(upper_tri_k > upper_tri_g)
    }
    
    return comparison, mi_gaussian_matrix, mi_ksg_matrix

# 使用示例
# comparison, mi_g, mi_k = compare_mi_methods(roi_ts)
# print(f"Gaussian MI mean: {comparison['mi_gaussian_mean']:.4f} bits")
# print(f"KSG MI mean: {comparison['mi_ksg_mean']:.4f} bits")
# print(f"KSG higher ratio: {comparison['ksg_higher_ratio']:.2%}")
```

- KSG MI 估计通常比 Gaussian MI 高 10-30% (捕获非线性关联)
- 若差异 > 30%，说明神经信号显著非高斯
- KSG 计算时间 ~2-3 倍于 Gaussian，但可接受

```python
#!/usr/bin/env python3
"""
fMRI η_IIT 分析流程 v2.0 - 采样策略模块
"""

import numpy as np

def determine_n_samples(n_regions, n_timepoints_available, method='adaptive'):
    """
    根据系统规模自适应确定采样数
    
    参数:
        n_regions: ROI 数量
        n_timepoints_available: 可用时间点总数
        method: 'adaptive' | 'fixed' | 'conservative'
    
    返回:
        n_samples: 建议采样数
    """
    if method == 'adaptive':
        # 自适应策略：N 越大，需要更多采样
        if n_regions <= 20:
            n_samples = min(200, n_timepoints_available)
        elif n_regions <= 50:
            n_samples = min(500, n_timepoints_available)
        elif n_regions <= 100:
            n_samples = min(1000, n_timepoints_available)
        else:
            n_samples = min(2000, n_timepoints_available)
    
    elif method == 'fixed':
        # 固定采样数 (DC-487 方案)
        n_samples = min(2000, n_timepoints_available)
    
    elif method == 'conservative':
        # 保守策略：KSG 收敛需要足够样本
        # 经验法则：n_samples > 10 × n_regions
        n_samples = min(max(500, 10 * n_regions), n_timepoints_available)
    
    return n_samples

def subsample_time_series(fmri_data, n_samples, method='stratified', seed=42):
    """
    从 fMRI 时间序列中采样
    
    参数:
        fmri_data: 原始时间序列 (n_timepoints, n_regions)
        n_samples: 目标采样数
        method: 'random' | 'sliding_window' | 'stratified'
        seed: 随机种子
    
    返回:
        samples: 采样后的时间序列 (n_samples, n_regions)
        indices: 使用的索引
    """
    np.random.seed(seed)
    n_timepoints = len(fmri_data)
    
    if n_samples >= n_timepoints:
        return fmri_data, np.arange(n_timepoints)
    
    if method == 'random':
        indices = np.random.choice(n_timepoints, n_samples, replace=False)
        indices = np.sort(indices)
    
    elif method == 'sliding_window':
        # 滑动窗口采样，保持时间连续性
        step = n_timepoints // n_samples
        indices = np.arange(0, n_timepoints, step)[:n_samples]
    
    elif method == 'stratified':
        # 分层采样：将时间序列分为 n_samples 段，每段取中点
        edges = np.linspace(0, n_timepoints, n_samples + 1, dtype=int)
        indices = (edges[:-1] + edges[1:]) // 2
    
    else:
        raise ValueError(f"Unknown sampling method: {method}")
    
    return fmri_data[indices], indices

def verify_sampling_convergence(fmri_data, n_regions, seed=42):
    """
    验证采样数收敛性
    
    返回:
        convergence_data: dict with η estimates at different sample sizes
    """
    sample_sizes = [200, 500, 1000, 2000, 4000]
    eta_estimates = []
    
    for n in sample_sizes:
        if n > len(fmri_data):
            break
        
        samples, _ = subsample_time_series(fmri_data, n, method='random', seed=seed)
        
        # 简化的η估计 (实际应使用完整流程)
        corr_matrix = np.corrcoef(samples.T)
        mi_approx = -0.5 * np.log(1 - corr_matrix**2 + 1e-10)
        phi_approx = np.sum(np.triu(mi_approx, k=1))
        
        # 归一化
        eta_approx = phi_approx / (n_regions * np.log2(n_regions) / 2)
        eta_estimates.append(eta_approx)
    
    # 计算收敛性
    if len(eta_estimates) >= 2:
        relative_change = abs(eta_estimates[-1] - eta_estimates[-2]) / eta_estimates[-2]
        converged = relative_change < 0.05  # 5% 阈值
    else:
        relative_change = None
        converged = False
    
    convergence_data = {
        'sample_sizes': sample_sizes[:len(eta_estimates)],
        'eta_estimates': eta_estimates,
        'relative_change': relative_change,
        'converged': converged
    }
    
    return convergence_data

# 
# convergence = verify_sampling_convergence(fmri_data, n_regions=100)
# print(f"Converged: {convergence['converged']} (relative change: {convergence['relative_change']:.2%})")
```

| N (ROI 数) | 建议采样数 | 理由 |
|-----------|-----------|------|

| 20 < N ≤ 50 | 500 | 中等规模 |
| 50 < N ≤ 100 | 1000 | 大规模，需要更多采样 |

## P1: η_IIT 符号严格化

### 符号系统澄清

**问题 (DC-487):** η_IIT 有时被当作函数η_IIT(N)，有时被当作常数η_IIT≈0.15

**严格化方案:**

| 符号 | 类型 | 定义 | 取值范围 |
|------|------|------|---------|
| η_IIT(N, P, T) | 函数 | 通用效率函数，依赖 N/平台/温度 | [0, 1] |
| η_IIT(P) | 特定值 | 特定平台在 N≈10, T→0 的效率 | 平台依赖 |
| η_IIT^0 | 常数 | 零温极限效率 (T→0) | ~0.15 |
| η_corr(N) | 函数 | 关联效率 = Φ_GHPII/Φ_IIT | [0, 1] |

**明确关系:**
```
η_corr(N) = η_IIT(N, P, T=0) × exp(-N/N_corr)

在 N≈10, T→0 极限:
η_corr(N≈10) ≈ η_IIT^0(P) × exp(-10/N_corr)

对于超导量子 (SC):
η_IIT^0(SC) ≈ 0.15±0.05
N_corr ≈ 8-12
→ η_corr(N=10) ≈ 0.15 × exp(-10/10) ≈ 0.15 × 0.37 ≈ 0.055
```

**修正后的预测 (D-487-01 澄清):**
```
D-487-01 (修订版): 
超导量子平台在 N=10-12 时，η_IIT^0(SC) ≈ 0.12-0.18

注意：这是η_IIT^0 (零温极限效率)，而非η_corr(N)
η_corr(N=10) ≈ η_IIT^0 × exp(-10/N_corr) ≈ 0.05-0.07
```

```python
def compute_phi_ghpii_from_counts(counts, n_qubits, k0=0.030, N_c=67):
    """
    从测量计数计算Φ_GHPII
    
    使用 T418 v4.1: Φ = k₀ × (N-1) / (1 + (N-1)/N_c)
    """
    # 计算 GHZ 态保真度
    p_000 = counts.get('0' * n_qubits, 0) / sum(counts.values())
    p_111 = counts.get('1' * n_qubits, 0) / sum(counts.values())
    p_ghz = p_000 + p_111
    
    # Φ_GHPII 正比于 GHZ 态概率
    phi = k0 * (n_qubits - 1) * p_ghz / (1 + (n_qubits - 1) / N_c)
    
    return phi, p_ghz

def compute_phi_iit(n_qubits):
    """
    计算 IIT 4.0 理论Φ上限
    
    使用近似：Φ_IIT(N) ≈ N × log₂(N) / 2
    """
    if n_qubits < 2:
        return 0
    return n_qubits * np.log2(n_qubits) / 2

def compute_eta_corr(phi_ghpii, phi_iit):
    """计算关联效率 η_corr = Φ_GHPII / Φ_IIT"""
    if phi_iit == 0:
        return 0
    return phi_ghpii / phi_iit

def run_ghz_simulation(n_qubits, shots=8192, noise_model=None):
    """
    运行 GHZ 态模拟
    
    返回:
        result: dict with phi_ghpii, phi_iit, eta_corr, p_ghz
    """
    # 创建电路
    qc = create_ghz_circuit(n_qubits)
    
    # 选择后端
    if noise_model is None:
        backend = Aer.get_backend('qasm_simulator')
    else:
        backend = Aer.get_backend('qasm_simulator')
    
    # 执行
    job = execute(qc, backend, shots=shots, noise_model=noise_model)
    counts = job.result().get_counts()
    
    # 计算Φ和η
    phi_ghpii, p_ghz = compute_phi_ghpii_from_counts(counts, n_qubits)
    phi_iit = compute_phi_iit(n_qubits)
    eta_corr = compute_eta_corr(phi_ghpii, phi_iit)
    
    return {
        'N': n_qubits,
        'Φ_GHPII': phi_ghpii,
        'Φ_IIT': phi_iit,
        'η_corr': eta_corr,
        'p_GHZ': p_ghz,
        'counts': counts
    }

def fit_eta_model(results):
    """
    拟合 η_corr(N) = η_IIT × exp(-N/N_corr)
    
    返回:
        params: (η_IIT, N_corr)
        covariance: 协方差矩阵
        r_squared: 拟合优度
    """
    N_array = np.array([r['N'] for r in results])
    eta_array = np.array([r['η_corr'] for r in results])
    
    def eta_model(N, eta_IIT, N_corr):
        return eta_IIT * np.exp(-N / N_corr)
    
    # 初始猜测
    p0 = [0.15, 10]
    
    # 拟合
    popt, pcov = curve_fit(eta_model, N_array, eta_array, p0=p0)
    eta_IIT_fit, N_corr_fit = popt
    
    # 计算 R²
    eta_pred = eta_model(N_array, *popt)
    ss_res = np.sum((eta_array - eta_pred) ** 2)
    ss_tot = np.sum((eta_array - np.mean(eta_array)) ** 2)
    r_squared = 1 - ss_res / ss_tot
    
    return {
        'eta_IIT': eta_IIT_fit,
        'N_corr': N_corr_fit,
        'eta_IIT_std': np.sqrt(pcov[0, 0]),
        'N_corr_std': np.sqrt(pcov[1, 1]),
        'r_squared': r_squared
    }
```
